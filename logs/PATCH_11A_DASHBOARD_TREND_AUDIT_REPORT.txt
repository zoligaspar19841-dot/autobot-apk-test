0001: class Trend(Widget):
0002:     def __init__(self, color=ORANGE, **kw):
0003:         super().__init__(**kw)
0004:         self.color = color
0005:         self.values = [random.uniform(40, 60) for _ in range(40)]
0006:         self.bind(pos=self.draw, size=self.draw)
0007: 
0008:     def set_values(self, vals):
0009:         if vals:
0010:             self.values = vals[-50:]
0011:         self.draw()
0012: 
0013:     def draw(self, *a):
0014:         self.canvas.clear()
0015:         with self.canvas:
0016:             Color(0.035, 0.04, 0.045, 1)
0017:             RoundedRectangle(pos=self.pos, size=self.size, radius=[18])
0018:             if len(self.values) < 2:
0019:                 return
0020:             mn, mx = min(self.values), max(self.values)
0021:             span = max(mx - mn, 0.000001)
0022:             pts = []
0023:             for i, v in enumerate(self.values):
0024:                 x = self.x + 8 + i * ((self.width - 16) / (len(self.values) - 1))
0025:                 y = self.y + 8 + ((v - mn) / span) * (self.height - 16)
0026:                 pts += [x, y]
0027:             Color(*self.color)
0028:             Line(points=pts, width=2.2)

0001: class TrendCanvasWidget(Widget):
0002:     def __init__(self, **kwargs):
0003:         super().__init__(**kwargs)
0004:         self.points = []
0005:         self.selected_index = -1
0006:         self.bind(pos=self.redraw, size=self.redraw)
0007: 
0008:     def set_chart(self, chart):
0009:         try:
0010:             self.points = chart.get('points') or []
0011:             self.selected_index = int(chart.get('selected_index', -1) or -1)
0012:         except Exception:
0013:             self.points = []
0014:             self.selected_index = -1
0015:         self.redraw()
0016: 
0017:     def redraw(self, *args):
0018:         self.canvas.clear()
0019:         with self.canvas:
0020:             Color(0.10, 0.12, 0.16, 1)
0021:             Rectangle(pos=self.pos, size=self.size)
0022: 
0023:             if not self.points:
0024:                 Color(0.5, 0.5, 0.5, 1)
0025:                 Line(points=[self.x, self.center_y, self.right, self.center_y], width=1)
0026:                 return
0027: 
0028:             vals = []
0029:             for p in self.points:
0030:                 try:
0031:                     vals.append(float(p.get('value') or 0))
0032:                 except Exception:
0033:                     vals.append(0.0)
0034: 
0035:             if not vals:
0036:                 return
0037: 
0038:             mn = min(vals)
0039:             mx = max(vals)
0040:             span = mx - mn if mx != mn else 1.0
0041: 
0042:             pad = 8
0043:             w = max(1, self.width - pad * 2)
0044:             h = max(1, self.height - pad * 2)
0045: 
0046:             coords = []
0047:             n = len(vals)
0048:             for i, v in enumerate(vals):
0049:                 x = self.x + pad + (i / max(1, n - 1)) * w
0050:                 y = self.y + pad + ((v - mn) / span) * h
0051:                 coords.extend([x, y])
0052: 
0053:             Color(0.95, 0.72, 0.15, 1)
0054:             if len(coords) >= 4:
0055:                 Line(points=coords, width=1.4)
0056: 
0057:             # baseline
0058:             Color(0.35, 0.35, 0.35, 1)
0059:             Line(points=[self.x + pad, self.y + pad, self.right - pad, self.y + pad], width=1)
0060: 
0061:             # selected crosshair
0062:             idx = self.selected_index
0063:             if idx < 0 or idx >= n:
0064:                 idx = n - 1
0065: 
0066:             sx = self.x + pad + (idx / max(1, n - 1)) * w
0067:             Color(0.2, 0.8, 1.0, 1)
0068:             Line(points=[sx, self.y + pad, sx, self.top - pad], width=1)
0069: 
0070:     def on_touch_down(self, touch):

0072:             return False
0073: 
0074:         if self.points:
0075:             ratio = (touch.x - self.x) / max(1, self.width)

0010:             self.points = chart.get('points') or []

0013:             self.points = []

0017:     def redraw(self, *args):
0018:         self.canvas.clear()
0019:         with self.canvas:
0020:             Color(0.10, 0.12, 0.16, 1)
0021:             Rectangle(pos=self.pos, size=self.size)
0022: 
0023:             if not self.points:
0024:                 Color(0.5, 0.5, 0.5, 1)
0025:                 Line(points=[self.x, self.center_y, self.right, self.center_y], width=1)
0026:                 return
0027: 
0028:             vals = []
0029:             for p in self.points:
0030:                 try:
0031:                     vals.append(float(p.get('value') or 0))
0032:                 except Exception:
0033:                     vals.append(0.0)
0034: 
0035:             if not vals:
0036:                 return
0037: 
0038:             mn = min(vals)
0039:             mx = max(vals)
0040:             span = mx - mn if mx != mn else 1.0
0041: 
0042:             pad = 8
0043:             w = max(1, self.width - pad * 2)
0044:             h = max(1, self.height - pad * 2)
0045: 
0046:             coords = []
0047:             n = len(vals)
0048:             for i, v in enumerate(vals):
0049:                 x = self.x + pad + (i / max(1, n - 1)) * w
0050:                 y = self.y + pad + ((v - mn) / span) * h
0051:                 coords.extend([x, y])
0052: 
0053:             Color(0.95, 0.72, 0.15, 1)
0054:             if len(coords) >= 4:
0055:                 Line(points=coords, width=1.4)
0056: 
0057:             # baseline
0058:             Color(0.35, 0.35, 0.35, 1)
0059:             Line(points=[self.x + pad, self.y + pad, self.right - pad, self.y + pad], width=1)
